# Install
````bash
npm install drawing-for-react
````
````bash
yarn add drawing-for-react
````
# Using
````js
import { Drawing } from 'drawing-for-react';

return (
  <Drawing
    background="#000000"
  />
)
````
## Clear
````js
import { clearDrawing } from 'drawing-for-react';

return (
  <button onClick={clearDrawing}>
    Clear
  </button>
)
````
